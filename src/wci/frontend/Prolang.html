start ::= <MINUS> <MODULE> <OPEN_PAREN> <IDENTIFIER> <CLOSE_PAREN> <DOT> statement_list <EOF>
statement_list ::= ( statement )*
statement ::= ( declaration_statement <DOT> | assignment_statement <DOT> | if_statement | for_statement | println | print | function_call <DOT> )
declaration_statement ::= ( variable_declaration | function_declaration )
variable_declaration ::= ( type ) identifier
function_declaration ::= <FUNC> identifier <OPEN_PAREN> parameterList <CLOSE_PAREN> <MINUS> <GT> returnType block
parameterList ::= ( parameter ( <COMMA> parameter )* )?
parameter ::= ( type ) ( <MULTIPLY> )? identifier
function_call ::= identifier <OPEN_PAREN> ( expression ( <COMMA> expression )* )? <CLOSE_PAREN>
block ::= <OPEN_BRACE> statement_list <CLOSE_BRACE>
if_statement ::= <IF> expression <THEN> block ( else_statement )?
else_statement ::= <ELSE> ( if_statement | block )
assignment_statement ::= identifier <EQ> ( function_call | expression )
for_statement ::= <FOR> for_header block
for_header ::= ( ( assignment_statement <DOT> expression <DOT> assignment_statement <DOT> ) | expression )

term ::= operand ( <MULTIPLY> operand | <MODULO> operand | <DIVIDE> operand )*
expressionList ::= expression ( <COMMA> expression )*
println ::= "io:fwriteln" <OPEN_PAREN> expression <CLOSE_PAREN> <DOT>
